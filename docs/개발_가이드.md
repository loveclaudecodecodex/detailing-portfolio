## 시작 직후 해야 할 세팅 (Bootstrap)

### 1. Lenis 부드러운 스크롤 적용

`src/components/lenis-provider.tsx`:
```tsx
"use client";
import { useEffect } from "react";
import Lenis from "lenis";

export function LenisProvider({ children }: { children: React.ReactNode }) {
  useEffect(() => {
    const lenis = new Lenis({ duration: 1.2, smoothWheel: true });
    function raf(time: number) { lenis.raf(time); requestAnimationFrame(raf); }
    const id = requestAnimationFrame(raf);
    return () => { cancelAnimationFrame(id); lenis.destroy(); };
  }, []);
  return <>{children}</>;
}
```

`src/app/layout.tsx` 에서 `<body>` 안을 `<LenisProvider>` 로 감싸기.

### 2. GSAP 초기화 (한 번만)

`src/lib/gsap.ts`:
```ts
import gsap from "gsap";
import { ScrollTrigger } from "gsap/ScrollTrigger";
import { SplitText } from "gsap/SplitText";
import { CustomEase } from "gsap/CustomEase";

if (typeof window !== "undefined") {
  gsap.registerPlugin(ScrollTrigger, SplitText, CustomEase);
  CustomEase.create("luxury", "M0,0 C0.16,1 0.3,1 1,1");
}

export { gsap, ScrollTrigger, SplitText, CustomEase };
```

### 3. 폰트 적용

`layout.tsx` 에 Pretendard CSS import 또는 next/font/local:
```tsx
import "pretendard/dist/web/variable/pretendardvariable.css";
```

영문 세리프(Cormorant Garamond) 필요 시 `next/font/google` 추가.

## WOW 모먼트 구현 예제

### A. 스크롤 풀스크린 변신 (히어로 직후)

`src/components/sections/wow-transform.tsx`:
```tsx
"use client";
import { useRef } from "react";
import Image from "next/image";
import { useGSAP } from "@gsap/react";
import { gsap, ScrollTrigger } from "@/lib/gsap";

export function WowTransform() {
  const sectionRef = useRef<HTMLElement>(null);
  const afterRef = useRef<HTMLDivElement>(null);

  useGSAP(() => {
    gsap.fromTo(
      afterRef.current,
      { clipPath: "inset(0 100% 0 0)" },
      {
        clipPath: "inset(0 0% 0 0)",
        ease: "none",
        scrollTrigger: {
          trigger: sectionRef.current,
          start: "top top",
          end: "+=150%",      // 1.5 화면 분량 스크롤
          scrub: 0.8,
          pin: true,
        },
      },
    );
  }, { scope: sectionRef });

  return (
    <section ref={sectionRef} className="relative h-screen overflow-hidden">
      <Image src="/images/wow-before.jpg" alt="시공 전" fill className="object-cover" priority />
      <div ref={afterRef} className="absolute inset-0">
        <Image src="/images/wow-after.jpg" alt="시공 후" fill className="object-cover" priority />
      </div>
      <h2 className="absolute bottom-12 left-1/2 -translate-x-1/2 text-white text-3xl md:text-5xl">
        스크롤로 만나는 시공 전·후
      </h2>
    </section>
  );
}
```

**주의:**
- 두 이미지는 동일 차종·각도여야 효과 극대
- 모바일에선 `end: "+=80%"` 정도로 줄이기
- `pin: true` 사용 시 부모에 `overflow-hidden` 필수

### C. 시공 타임랩스 + 스크롤 컨트롤

```tsx
"use client";
import { useRef } from "react";
import { useGSAP } from "@gsap/react";
import { gsap, ScrollTrigger } from "@/lib/gsap";

export function WowTimelapse() {
  const sectionRef = useRef<HTMLElement>(null);
  const videoRef = useRef<HTMLVideoElement>(null);

  useGSAP(() => {
    const video = videoRef.current;
    if (!video) return;

    const onReady = () => {
      const duration = video.duration; // 30초
      ScrollTrigger.create({
        trigger: sectionRef.current,
        start: "top top",
        end: "+=200%",        // 2 화면 스크롤 = 30초 영상
        scrub: 0.5,
        pin: true,
        onUpdate: (self) => {
          video.currentTime = duration * self.progress;
        },
      });
    };

    if (video.readyState >= 1) onReady();
    else video.addEventListener("loadedmetadata", onReady, { once: true });
  }, { scope: sectionRef });

  return (
    <section ref={sectionRef} className="relative h-screen overflow-hidden bg-brand-black">
      <video
        ref={videoRef}
        src="/videos/process-timelapse.mp4"
        muted
        playsInline
        preload="auto"
        className="absolute inset-0 w-full h-full object-cover"
      />
      <div className="absolute bottom-12 left-1/2 -translate-x-1/2 text-white">
        <p className="text-xl md:text-3xl">12시간의 정성, 30초에 담다</p>
      </div>
    </section>
  );
}
```

**주의:**
- iOS Safari에선 `muted playsInline` 필수
- `preload="auto"` 라도 데이터 무거우면 LCP 영향 — `intersection observer` 로 트리거 직전 로딩
